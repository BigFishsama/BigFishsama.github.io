git config --global user.name "你的名字"
git config --global user.email "邮箱地址"

mkdir xxx
cd xxx
git init

# 把本地 master 改名为 main，因为github上默认分支叫main
git branch -m master main   
# 关联远端仓库地址，后续就是origin代指
git remote add origin https://github.com/用户名/仓库名.git
# 克隆仓库
git clone https://github.com/用户名/仓库名.git 

git status           #查看仓库状态
git add xxx.x        #添加文件到缓存区
git add .            #将工作区所有文件添加入缓存区
git commit -m "xxx"  #将缓冲区文件提交到版本库
git log              #查看提交记录
git push -u origin main   #前文我们已将本地master改成main，这是为了本地和远端同步名称，防止新建
git pull             #将本地当前的分支按照名称合并到远端

#0x01 Git分支

以下内容暂时可以忽略

为了统一管理，采用以下三种命名分支：

  • master(main)：稳定的正式版
  • develop：从master分支拓展，再与feature分支合并，作为经常变动代码的测试版，无误之后合并入master中
  • feature：从develop分支拓展，作为开发拓展功能的分支，合并入develop分支，废弃时直接弃用整个分支即可
